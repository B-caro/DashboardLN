import numpy as np

num_aleatorios = np.random.randint(0, 100, (6, 6))

clone = num_aleatorios.copy()

diagonal = np.diag(clone)

print('TÃº carton de juego:')

print(np.diag(np.fliplr(clone)))

print(clone[:, 0])
